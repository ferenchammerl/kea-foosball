CREATE TABLE users
(
ID int NOT NULL AUTO_INCREMENT,
USERNAME varchar(31)  NOT NULL,
PASSWORD char(64) NOT NULL,
SALT char(16)  NOT NULL,
EMAIL varchar(255) NOT NULL,
ISHOMEN BOOLEAN DEFAULT FALSE,
ELO integer DEFAULT '1000',
PRIMARY KEY (ID),
UNIQUE  (USERNAME),
UNIQUE  (EMAIL)
);

CREATE TABLE moderators
(
ID int NOT NULL AUTO_INCREMENT,
USERNAME varchar(31)  NOT NULL,
PASSWORD char(64) NOT NULL,
SALT char(16)  NOT NULL,
EMAIL varchar(255) NOT NULL,
PRIMARY KEY (ID),
UNIQUE  (USERNAME),
UNIQUE  (EMAIL)
);





CREATE TABLE games 
(
ID int NOT NULL AUTO_INCREMENT,
LOCATION varchar(255)  DEFAULT 'Unknown',
WIN1 int NOT NULL,
WIN2 int NOT NULL,
LOS1 int NOT NULL,
LOS2 int NOT NULL,
CON1 BOOLEAN DEFAULT FALSE,
CON2 BOOLEAN DEFAULT FALSE,
CON3 BOOLEAN DEFAULT FALSE,
CON4 BOOLEAN DEFAULT FALSE,
PRIMARY KEY (ID)
);
